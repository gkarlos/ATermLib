51  std::size_t get_sufficiently_large_postfix_index(const std::string& prefix_)
52  {
53  std::size_t index=0;
54  for(const detail::_function_symbol& f: function_symbol::function_symbol_store())
55  {
56  const std::string& function_name=f.first.name();
57 
58  if (function_name.compare(0,prefix_.size(),prefix_)==0) // The function name starts with the prefix
59  {
60  std::string potential_number=function_name.substr(prefix_.size()); // Get the trailing string after prefix_ of function_name.
61  std::size_t end_of_number;
62  try
63  {
64  std::size_t number=std::stol(potential_number,&end_of_number);
65  if (end_of_number==potential_number.size()) // A proper number was read.
66  {
67  if (number>=index)
68  {
69  index=number+1;
70  }
71  }
72  }
73  catch (std::exception&)
74  {
75  // Can be std::invalid_argument or an out_of_range exception.
76  // In both cases nothing needs to be done, and the exception can be ignored.
77  }
78  }
79 
80  }
81  return index;
82  }
