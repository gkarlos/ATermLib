61 static void write_string_with_escape_symbols(const std::string& s, std::ostream& os)
62 {
63  // Check whether the string starts with a - or a number, or contains the symbols
64  // \, ", (, ), [, ], comma, space, \t, \n or \r. If yes, the string will be
65  // surrounded by quotes, and the symbols \, ", \t, \n, \r
66  // will be preceded by an escape symbol.
67 
68  // assert(s.size()>0);
69  char c=s[0];
70  bool contains_special_symbols= c=='-' || isdigit(c);
71 
72  for(std::string::const_iterator i=s.begin(); !contains_special_symbols && i!=s.end(); ++i)
73  {
74  if (*i=='\\' || *i=='"' || *i=='(' || *i==')' || *i=='[' || *i==']' || *i==',' || *i==' ' || *i=='\n' || *i=='\t' || *i=='\r')
75  {
76  contains_special_symbols=true;
77  }
78  }
79 
80  if (contains_special_symbols)
81  {
82  /* This function symbol needs quotes */
83  os << "\"";
84  for(std::string::const_iterator i=s.begin(); i!=s.end(); ++i)
85  {
86  /* We need to escape special characters */
87  switch (*i)
88  {
89  case '\\':
90  case '"':
91  os << "\\" << *i;
92  break;
93  case '\n':
94  os << "\\n";
95  break;
96  case '\t':
97  os << "\\t";
98  break;
99  case '\r':
100  os << "\\r";
101  break;
102  default:
103  os << *i;
104  break;
105  }
106  }
107  os << "\"";
108  }
109  else
110  {
111  os << s;
112  }
113 }

115 static void writeToStream(const aterm& t, std::ostream& os)
116 {
117  if (t.type_is_int())
118  {
119  os << aterm_int(t).value();
120  }
121  else if (t.type_is_list())
122  {
123  os << "[";
124  const aterm_list& list = down_cast<aterm_list>(t);
125  for(aterm_list::const_iterator i=list.begin(); i!=list.end(); ++i)
126  {
127  if (i!=list.begin())
128  {
129  os << ",";
130  }
131  writeToStream(*i, os);
132  }
133  os << "]";
134  }
135  else // t.type_is_appl()
136  {
137  const aterm_appl& appl = down_cast<aterm_appl>(t);
138  const function_symbol& sym = appl.function();
139  write_string_with_escape_symbols(sym.name(),os);
140  if (sym.arity() > 0)
141  {
142  os << "(";
143  writeToStream(appl[0], os);
144  for (std::size_t i = 1; i < sym.arity(); i++)
145  {
146  os << ",";
147  writeToStream(appl[i], os);
148  }
149  os << ")";
150  }
151  }
152 }

176 {

237 static aterm_list fparse_terms(int* c, istream& is)
238 {
239  if (*c==']' || *c==')') // The termlist must be empty.
240  {
241  return aterm_list();
242  }
243 
244  aterm el = fparse_term(c, is);
245  aterm_list list;
246  list.push_front(el);
247 
248  while (*c == ',')
249  {
250  fnext_skip_layout(c, is);
251  el = fparse_term(c, is);
252  list.push_front(el);
253  }
254  return reverse(list);
255 }

261 static string fparse_quoted_string(int* c, istream& is)
262 {
263  /* We need a buffer for printing and parsing */
264  std::string function_string;
265 
266  /* First parse the identifier */
267  fnext_char(c, is);
268 
269  while (*c != '"')
270  {
271  switch (*c)
272  {
273  case EOF:
274  throw atermpp::runtime_error("Premature end of file while parsing quoted function symbol.");
275  case '\\':
276  fnext_char(c, is);
277  if (*c == EOF)
278  {
279  throw atermpp::runtime_error("Premature end of file while parsing quoted function symbol.");
280  }
281  switch (*c)
282  {
283  case 'n':
284  function_string+='\n';
285  break;
286  case 'r':
287  function_string+='\r';
288  break;
289  case 't':
290  function_string+= '\t';
291  break;
292  default:
293  function_string+= *c;
294  break;
295  }
296  break;
297  default:
298  function_string+= *c;
299  break;
300  }
301  fnext_char(c, is);
302  }
303  fnext_skip_layout(c, is);

326 static aterm_appl parse_arguments(const string& f, int *c, istream& is)
327 {
328  /* Time to parse the arguments */
329  aterm_list args;
330  if (*c == '(')
331  {
332  fnext_skip_layout(c, is);
333  if (*c != ')')
334  {
335  args = fparse_terms(c, is);
336  }
337 
338  if (*c != ')')
339  {
340  throw atermpp::runtime_error("Missing ')' while parsing term");
341  }
342  fnext_skip_layout(c, is);
343  }
344 
345  /* Wrap up this function application */
346  const function_symbol sym(f, args.size());
347  return aterm_appl(sym, args.begin(), args.end());

357 {

382 static aterm fparse_term(int* c, istream& is)
383 {
384  switch (*c)
385  {
386  case '"':
387  {
388  string f=fparse_quoted_string(c, is);
389  return parse_arguments(f,c,is);
390  }
391  case '[':
392  {
393  fnext_skip_layout(c, is);
394  if (*c == ']')
395  {
396  fnext_skip_layout(c, is);
397  return aterm_list(); // The empty list has just been parsed.
398  }
399  else
400  {
401  const aterm result = fparse_terms(c, is);
402  if (*c != ']')
403  {
404  throw atermpp::runtime_error("Expecting ']' while parsing term");
405  }
406  fnext_skip_layout(c, is);
407  return result;
408  }
409  }
410  default:
411  {
412  if (isdigit(*c) || *c == '-')
413  {
414  return fparse_num(c, is);
415  }
416  string f=fparse_unquoted_string(c, is);
417  return parse_arguments(f,c,is);
418  }
419  }
