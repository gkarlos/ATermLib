94 // Count the total number of occurrences of function symbols t where
95 // t is viewed as a shared term. The result is stored in the unordered_map count.
96 // Visited is used to avoid visiting terms too often.
97 static std::unordered_map<function_symbol, std::size_t> count_the_number_of_unique_function_symbols_in_a_term(
98  const aterm& t)
99 {
100  std::stack<aterm> todo_stack({t});
101  std::unordered_set<aterm> visited({t});
102  std::unordered_map<function_symbol, std::size_t> count;
103 
104  while (!todo_stack.empty())
105  {
106  const aterm t=todo_stack.top();
107  todo_stack.pop();
108  assert(visited.count(t)>0);
109 
110  if (t.type_is_int())
111  {
112  count[detail::function_adm.AS_INT]++;
113  }
114  else if (t.type_is_list())
115  {
116  const aterm_list& list = down_cast<const aterm_list>(t);
117  if (list==aterm_list())
118  {
119  count[detail::function_adm.AS_EMPTY_LIST]++;
120  }
121  else
122  {
123  count[detail::function_adm.AS_LIST]++;
124  if (visited.count(list.tail())==0)
125  {
126  visited.insert(list.tail());
127  todo_stack.push(list.tail());
128  }
129  if (visited.count(list.front())==0)
130  {
131  visited.insert(list.front());
132  todo_stack.push(list.front());
133  }
134  }
135  }
136  else
137  {
138  const aterm_appl& ta=down_cast<aterm_appl>(t);
139  function_symbol sym = ta.function();
140  count[sym]++;
141  for (std::size_t i=ta.size(); i>0; )
142  {
143  --i;
144  if (visited.count(ta[i])==0)
145  {
146  visited.insert(ta[i]);
147  todo_stack.push(ta[i]);
148  }
149  }
150  }
151  }
152  return count;
153 }

238 static void writeBits(std::size_t val, const std::size_t nr_bits, ostream& os)
239 {
240  for (std::size_t cur_bit=0; cur_bit<nr_bits; cur_bit++)
241  {
242  bit_buffer <<= 1;
243  bit_buffer |= (val & 0x01);
244  val >>= 1;
245  if (++bits_in_buffer == 8)
246  {
247  os.put(bit_buffer);
248  bits_in_buffer = 0;
249  bit_buffer = '\0';
250  }
251  }
252  assert(val==0);
253 }

279 static
280 bool readBits(std::size_t& val, const std::size_t nr_bits, istream& is)
281 {
282  std::size_t cur_bit, mask = 1;
283 
284  val = 0;
285  for (cur_bit=0; cur_bit<nr_bits; cur_bit++)
286  {
287  if (bits_in_buffer == 0)
288  {
289  int byte = is.get();
290  if (is.fail())
291  {
292  return false;
293  }
294  bit_buffer = (unsigned char)byte;
295  bits_in_buffer = 8;
296  }
297  val |= (bit_buffer & 0x80 ? mask : 0);
298  mask <<= 1;
299  bit_buffer <<= 1;
300  bits_in_buffer--;
301  }
302  return true;
303 }

406 static void build_arg_tables(const std::unordered_map<function_symbol, std::size_t>& index,
407  std::vector<sym_write_entry>& sym_entries)
408 {
409  for (sym_write_entry& cur_entry: sym_entries)
410  {
411  std::size_t arity = cur_entry.id.arity();
412  cur_entry.top_symbols = std::vector<top_symbols_t>(arity);
413 
414  if (cur_entry.id!=detail::function_adm.AS_INT)
415  {
416  for (std::size_t cur_arg=0; cur_arg<arity; cur_arg++)
417  {
418  top_symbols_t& tss = cur_entry.top_symbols[cur_arg];
419  std::size_t total_top_symbols = 0;
420  for(const pair<aterm, std::size_t>& p: cur_entry.write_terms)
421  {
422  aterm term = p.first;
423  aterm arg;
424  if (term.type_is_list())
425  {
426  aterm_list list(term);
427  assert(list!=aterm_list());
428  assert(arity == 2);
429  if (cur_arg == 0)
430  {
431  arg = list.front();
432  }
433  else
434  {
435  arg = list.tail();
436  }
437  }
438  else if (term.type_is_appl())
439  {
440  arg = down_cast<const aterm_appl>(term)[cur_arg];
441  }
442  else
443  {
444  throw mcrl2::runtime_error("Internal inconsistency found in internal data structure. Illegal term.");
445  }
446  sym_write_entry* topsym = get_top_symbol(arg, index, sym_entries);
447 
448  if (tss.index_into_symbols.count(topsym->id)==0)
449  {
450  total_top_symbols++;
451  std::size_t index=tss.symbols.size();
452  tss.symbols.emplace_back(topsym-&sym_entries[0], index);
453  tss.index_into_symbols[topsym->id]=index;
454  }
455  }
456  tss.code_width=bit_width(total_top_symbols);
457  }
458  }
459  }
460 }

508 {

510  std::unordered_set<aterm> visited;
511  stack.emplace(t, index, sym_entries);
512 
513  do
514  {
515  write_todo& current=stack.top();

578 static void write_term(const aterm& t, const std::unordered_map<function_symbol, std::size_t>& index, ostream& os, std::vector<sym_write_entry>& sym_entries)
579 {
580  std::stack<write_todo> stack;
581  stack.emplace(t, index, sym_entries);
582 
583  do
584  {
585  write_todo& current = stack.top();
586 
587  if (current.term.type_is_int())
588  {
589  // If aterm integers are > 32 bits, then they cannot be read on a 32 bit machine.
590  writeBits(aterm_int(current.term).value(), INT_SIZE_IN_BAF, os);
591  }
592  else
593  if (current.arg < current.entry->id.arity())
594  {
595  write_todo item(subterm(current.term, current.arg), index, sym_entries);
596 
597  const top_symbol& ts = find_top_symbol(&current.entry->top_symbols[current.arg], item.entry->id);
598  writeBits(ts.code, current.entry->top_symbols[current.arg].code_width, os);
599  sym_write_entry& arg_sym = sym_entries[ts.index];
600  std::size_t arg_trm_idx = arg_sym.write_terms.at(item.term);
601  writeBits(arg_trm_idx, arg_sym.term_width, os);
602 
603  ++current.arg;
604 
605  if (arg_trm_idx >= arg_sym.cur_index)
606  {
607  stack.push(item);
608  }
609  continue;
610  }
611 
612  ++current.entry->cur_index;
613  stack.pop();
614  }
615  while (!stack.empty());
616  flushBitsToWriter(os);
617 }

620 static void write_baf(const aterm& t, ostream& os)
621 {
622  /* Initialize bit buffer */
623  bit_buffer = '\0';
624  bits_in_buffer = 0; /* how many bits in bit_buffer are used */
625 
626  std::unordered_map<function_symbol, std::size_t> index;
627  std::unordered_map<function_symbol, std::size_t> count=count_the_number_of_unique_function_symbols_in_a_term(t);
628  std::size_t nr_unique_symbols = count.size();
629 
630  std::vector<sym_write_entry> sym_entries;
631  sym_entries.reserve(nr_unique_symbols);
632 
633  /* Collect all unique symbols in the input term */
634 
635  std::size_t cur=0;
636  for(const pair<const function_symbol, std::size_t>& p: count)
637  {
638  const function_symbol sym=p.first;
639  const std::size_t nr_of_occurrences=p.second;
640 
641  sym_entries.emplace_back(sym, bit_width(nr_of_occurrences));
642  index[sym] = cur;
643 
644  cur++;
645  }
646 
647  assert(cur == nr_unique_symbols);
648 
649  collect_terms(t, index, sym_entries);
650  build_arg_tables(index, sym_entries);
651 
652  /* write header */
653 
654  writeInt(0, os);
655  writeInt(BAF_MAGIC, os);
656  writeInt(BAF_VERSION, os);
657  writeInt(nr_unique_symbols, os);
658  write_all_symbols(os, sym_entries);
659 
660  /* Write the top symbol */
661  writeInt(get_top_symbol(t,index,sym_entries)-&sym_entries[0], os);
662 
663  write_term(t, index, os, sym_entries);
664 }

761  // AS_INT is registered as having 1 argument, but that needs to be retrieved in a special way.
762  if (current.sym->sym != detail::function_adm.AS_INT && current.args.size() < current.sym->sym.arity())
763  {
764  if (readBits(value, current.sym->sym_width[current.args.size()], is) &&
765  value < current.sym->topsyms[current.args.size()].size())
766  {
767  sym_read_entry* arg_sym = &read_symbols[current.sym->topsyms[current.args.size()][value]];
768  if (readBits(value, arg_sym->term_width, is) &&
769  value < arg_sym->terms.size())
770  {
771  current.callresult = &arg_sym->terms[value];
772  if (!current.callresult->defined())
773  {
774  stack.emplace(arg_sym, current.callresult);
775  }
776  continue;
777  }
778  }
779  throw mcrl2::runtime_error("Could not read valid aterm from stream.");
780  }
781 
782  if (current.sym->sym == detail::function_adm.AS_INT)
783  {
784  if (readBits(value, INT_SIZE_IN_BAF, is))
785  {
786  *current.result = aterm_int(value);
787  }
788  }
789  else if (current.sym->sym==detail::function_adm.AS_EMPTY_LIST)
790  {
791  *current.result = aterm_list();
792  }
793  else if (current.sym->sym == detail::function_adm.AS_LIST)
794  {
795  aterm_list result = atermpp::down_cast<aterm_list>(current.args[1]);
796  result.push_front(current.args[0]);
797  *current.result = result;
798  }
799  else // sym is a function application
800  {
801  *current.result = aterm_appl(current.sym->sym, current.args.begin(), current.args.end());
802  }
803  stack.pop();
804  }
805  while (!stack.empty());
806 
807  return result;
808 }

815 aterm read_baf(istream& is)
816 {
817  // Initialize bit buffer
818  bit_buffer = '\0';
819  bits_in_buffer = 0; // how many bits in bit_buffer are used
820 
821  // Read header
822  std::size_t val = readInt(is);
823  if (val == 0)
824  {
825  val = readInt(is);
826  }
827  if (val != BAF_MAGIC)
828  {
829  throw mcrl2::runtime_error("Error while reading file: The file is not correct as it does not have the BAF_MAGIC control sequence at the right place.");
830  }
831 
832  std::size_t version = readInt(is);
833  if (version != BAF_VERSION)
834  {
835  throw mcrl2::runtime_error("The BAF version (" + std::to_string(version) + ") of the input file is incompatible with the version (" + std::to_string(BAF_VERSION) +
836  ") of this tool. The input file must be regenerated. ");
837  }
838 
839  std::size_t nr_unique_symbols = readInt(is);
840 
841  // Allocate symbol space
842  std::vector<sym_read_entry> read_symbols(nr_unique_symbols);
843 
844  read_all_symbols(is, nr_unique_symbols, read_symbols);
845 
846  val = readInt(is);
847  aterm result=read_term(&read_symbols[val], is, read_symbols);
848  return result;
849 }
