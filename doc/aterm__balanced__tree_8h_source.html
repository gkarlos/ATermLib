57  detail::_aterm_appl<aterm>* make_tree(ForwardTraversalIterator& p, const std::size_t size, const Transformer& transformer )
58  {
59  if (size>1)
60  {
61  std::size_t left_size = (size + 1) >> 1; // size/2 rounded up.
62  const term_balanced_tree left_tree(make_tree(p, left_size,transformer));

265  void increment()
266  {
267 
268  --m_top_of_stack;
269 
270  if (m_top_of_stack>0)
271  {
272  detail::_aterm* current = m_stack[m_top_of_stack-1];
273 
274  if (current->function()!=term_balanced_tree < Term >::tree_node_function())
275  {
276  return;
277  }
278  --m_top_of_stack;
279 
280  do
281  {
282  m_stack[m_top_of_stack++]=atermpp::detail::address((reinterpret_cast<detail::_aterm_appl<aterm> *>(current)->arg[1]));
283  current=atermpp::detail::address(reinterpret_cast<detail::_aterm_appl<aterm> *>(current)->arg[0]);
284  }
