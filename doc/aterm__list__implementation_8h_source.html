37 template <typename Term>
38 inline
39 term_list<Term> push_back(const term_list<Term>& l, const Term& el)
40 {
41  typedef typename term_list<Term>::const_iterator const_iterator;
42 
43  const std::size_t len = l.size();
44  if (len<max_len_of_short_list)
45  {
46  // The list is short, use the stack for temporal storage.
47  const_iterator* buffer = MCRL2_SPECIFIC_STACK_ALLOCATOR(const_iterator, len);
48 
49  /* Collect all elements of list in buffer */
50 
51  std::size_t j=0;
52  for (const_iterator i = l.begin(); i != l.end(); ++i, ++j)
53  {
54  buffer[j]=i;
55  }
56 
57  term_list<Term> result;
58  result.push_front(el);
59 
60  /* Insert elements at the front of the list */
61  while (j>0)
62  {
63  j=j-1;
64  result.push_front(*buffer[j]);
65  }
66  return result;
67  }
68  else
69  {
70  // The list is long. Use the heap to store intermediate data.
71  std::vector<Term> buffer;
72  buffer.reserve(len);
73 
74  for (const Term& t: l)
75  {
76  buffer.push_back(t);
77  }
78 
79  term_list<Term> result;
80  result.push_front(el);
81 
82  /* Insert elements at the front of the list */
83  for (typename std::vector<Term>::reverse_iterator i=buffer.rbegin(); i!=buffer.rend(); ++i)
84  {
85  result.push_front(*i);
86  }
87  return result;
88  }
89 }

105 template <typename Term>
106 inline
107 term_list<Term> remove_one_element(const term_list<Term>& list, const Term& t)
108 {
109  typedef typename term_list<Term>::const_iterator const_iterator;
110 
111  std::size_t len=0;
112  const_iterator i = list.begin();
113  for( ; i!=list.end(); ++i, ++len)
114  {
115  if (*i==t)
116  {
117  break;
118  }
119  }
120 
121  if (i==list.end())
122  {
123  // Term t not found in list.
124  return list;
125  }
126 
127  const_iterator* buffer = MCRL2_SPECIFIC_STACK_ALLOCATOR(const_iterator, len);
128 
129  term_list<Term> result = list;
130  std::size_t k=0;
131  for(const_iterator j = list.begin(); j != i; ++j, ++k)
132  {
133  buffer[k]=j;
134  result.pop_front();
135  }
136  assert(len==k);
137  assert(result.front()==t);
138  result.pop_front(); // skip the element.
139 
140  while (k>0)
141  {
142  k=k-1;
143  result.push_front(*buffer[k]);
144  }
145 
146  return result;
147 }
