114 inline std::size_t hash_number(detail::_aterm *t)
115 {
116  const function_symbol& f=t->function();
117  const std::hash<function_symbol> function_symbol_hasher;
118  std::size_t hnr = function_symbol_hasher(f);
119 
120  // If the term is an aterm_int, the argument is a value, which is hashed differently.
121  if (f==function_adm.AS_INT)
122  {
123  return detail::hash_value_aterm_int(*(reinterpret_cast<const std::size_t*>(t)+TERM_SIZE));
124  }
125  // Else treat the arguments in a normal way.
126  const std::size_t* begin=reinterpret_cast<const std::size_t*>(t)+TERM_SIZE;
127  const std::size_t* end=begin+f.arity();
128  std::hash<atermpp::detail::_aterm*> aterm_hasher;
129  for (const std::size_t* i=begin; i!=end; ++i)
130  {
131  hnr = COMBINE_aux(hnr, aterm_hasher(reinterpret_cast<const _aterm*>(*i)));
132  }
133 
134  return hnr;
135 }

137 inline _aterm* allocate_term(const std::size_t size)
138 {
139  assert(size>=TERM_SIZE);
140  if (size >= terminfo_size)
141  {
142  // Resize the size of terminfo to the minimum of twice its old size and size+1;
143  const std::size_t old_term_info_size=terminfo_size;
144  terminfo_size <<=1; // Multiply by 2.
145  if (size>=terminfo_size)
146  {
147  terminfo_size=size+1;
148  }
149  terminfo=reinterpret_cast<TermInfo*>(realloc(terminfo,terminfo_size*sizeof(TermInfo)));
150  if (terminfo==nullptr)
151  {
152  throw std::runtime_error("Out of memory. Failed to allocate an extension of terminfo.");
153  }
154  for(std::size_t i=old_term_info_size; i<terminfo_size; ++i)
155  {
156  new (&terminfo[i]) TermInfo();
157  }
158  assert(size<terminfo_size);
159  }
160 
161  if (total_nodes_in_hashtable>=aterm_table_size)
162  {
163  // The hashtable is not big enough to hold nr_of_nodes_for_the_next_garbage_collect. So, resizing
164  // is wise (although not necessary, due to the structure of the hastable, which allows is to contain
165  // an arbitrary number of element, at some performance penalty.
166  resize_aterm_hashtable();
167  }
168 
169  TermInfo& ti = terminfo[size];
170  if (garbage_collect_count_down>0)
171  {
172  garbage_collect_count_down--;
173  }
174 
175  if (garbage_collect_count_down==0 && ti.at_freelist==nullptr) // It is time to collect free terms, and there are
176  // no free terms left.
177  {
178  collect_terms_with_reference_count_0();
179  }
180  if (ti.at_freelist==nullptr)
181  {
182  /* there is no more memory of the current size allocate a block */
183  allocate_block(size);
184  assert(ti.at_block != nullptr);
185  }
186 
187  _aterm *at = ti.at_freelist;
188  ti.at_freelist = ti.at_freelist->next();
189  assert(at->reference_count_indicates_is_in_freelist());
190  at->reset_reference_count();
191  assert(ti.at_block != nullptr);
192  return at;
193 }

195 inline void remove_from_hashtable(_aterm *t)
196 {
197  /* Remove the node from the aterm_hashtable */
198  _aterm *prev=nullptr;
199  const std::size_t hnr = hash_number(t) & aterm_table_mask;
200  _aterm *cur = aterm_hashtable[hnr];
201 
202  do
203  {
204  assert(cur!=nullptr); /* This only occurs if the hashtable is in error. */
205  if (cur == t)
206  {
207  if (prev)
208  {
209  prev->set_next(cur->next());
210  }
211  else
212  {
213  aterm_hashtable[hnr] = cur->next();
214  }
215  /* Put the node in the appropriate free list */
216  total_nodes_in_hashtable--;
217  return;
218  }
219  }
220  while (((prev=cur), (cur=cur->next())));
221  assert(0);
222 }
