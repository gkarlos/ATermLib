170  void find_next_index()
171  {
172  std::vector<bdd_node> path_stack;
173  bdd_node node = m_subset->m_bdd_root;
174 
175  while (true)
176  {
177  assert(m_subset->m_set != nullptr);
178  if (m_index >= m_subset->m_set->size())
179  {
180  m_index = -1;
181  return;
182  }
183 
184  while (node.is_node())
185  {
186  path_stack.push_back(node);
187  node = (m_index & ((std::size_t)1 << node.bit())) ? node.true_node() : node.false_node();
188  }
189 
190  if (node.is_true())
191  {
192  return;
193  }
194 
195  while (true)
196  {
197  bdd_node start;
198  std::size_t bit;
199 
200  if (path_stack.empty())
201  {
202  start = m_subset->m_bdd_root;
203  bit = m_subset->m_bits;
204  }
205  else
206  {
207  start = node;
208  bit = path_stack.back().bit();
209  }
210 
211  if (!start.is_false())
212  {
213  assert(start.is_node());
214  bool found = false;
215  for (std::size_t i = start.bit() + 1; i < bit; i++)
216  {
217  if (!(m_index & ((std::size_t)1 << i)))
218  {
219  m_index |= ((std::size_t)1 << i);
220  m_index &= ~(((std::size_t)1 << i) - 1);
221  found = true;
222  break;
223  }
224  }
225  if (found)
226  {
227  break;
228  }
229  }
230 
231  if (path_stack.empty())
232  {
233  m_index = -1;
234  return;
235  }
236  else
237  {
238  node = path_stack.back();
239  path_stack.pop_back();
240  if (!(m_index & ((std::size_t)1 << bit)) && !node.true_node().is_false())
241  {
242  m_index |= ((std::size_t)1 << bit);
243  m_index &= ~(((std::size_t)1 << bit) - 1);
244  break;
245  }
246  }
247  }
248  }

288  {

315  }

319  }
320  }

323  completed &= ~(((std::size_t)1 << bit) - 1);
324  }
325  }
