28 template <class Term, class InputIterator, class ATermConverter>
29 _aterm* local_term_appl_with_converter(const function_symbol& sym,
30  const InputIterator begin,
31  const InputIterator end,
32  const ATermConverter& convert_to_aterm)
33 {
34  const std::size_t arity = sym.arity();
35 
36  const std::hash<function_symbol> function_symbol_hasher;
37  std::size_t hnr = function_symbol_hasher(sym);
38 
39  aterm* temporary_args = MCRL2_SPECIFIC_STACK_ALLOCATOR(aterm, arity);
40  std::size_t j=0;
41  for (InputIterator i=begin; i!=end; ++i, ++j)
42  {
43  new (&(temporary_args[j])) Term(convert_to_aterm(*i));
44  CHECK_TERM(temporary_args[j]);
45  hnr = COMBINE(hnr, temporary_args[j]);
46  }
47  assert(j==arity);
48 
49 
50  _aterm* cur = detail::aterm_hashtable[hnr& detail::aterm_table_mask];
51  while (cur)
52  {
53  if (cur->function()==sym)
54  {
55  bool found = true;
56  for (std::size_t i=0; i<arity; i++)
57  {
58  if (reinterpret_cast<_aterm_appl<Term>*>(cur)->arg[i] != temporary_args[i])
59  {
60  found = false;
61  break;
62  }
63  }
64  if (found)
65  {
66  for(std::size_t i=0; i<arity; ++i)
67  {
68  temporary_args[i].~aterm();
69  }
70  return cur;
71  }
72  }
73  cur = cur->next();
74  }
75 
76  assert(cur==nullptr);
77  detail::_aterm* new_term = (detail::_aterm_appl<Term>*) detail::allocate_term(TERM_SIZE_APPL(arity));
78 
79  // We copy the content of the temporary_args, without destruction/construction and adapting the reference counts.
80  for(std::size_t i=0; i<arity; ++i)
81  {
82  new (&(reinterpret_cast<detail::_aterm_appl<Term>*>(const_cast<detail::_aterm*>(new_term))->arg[i])) _aterm*(detail::address(temporary_args[i]));
83  }
84  new (&const_cast<detail::_aterm*>(const_cast<detail::_aterm*>(new_term))->function()) function_symbol(sym);
85 
86  // Apply the table mask after allocate_term, which may resize the table.
87  insert_in_hashtable(new_term,hnr& detail::aterm_table_mask);
88  call_creation_hook(new_term);
89 
90  return new_term;
91 }

154 template <class Term>
155 _aterm* term_appl1(const function_symbol& sym, const Term& arg0)
156 {
157  assert(sym.arity()==1);
158  CHECK_TERM(arg0);
159 
160  const std::hash<function_symbol> function_symbol_hasher;
161  std::size_t hnr = COMBINE(function_symbol_hasher(sym), arg0);
162 
163  _aterm *cur = detail::aterm_hashtable[hnr & detail::aterm_table_mask];
164  while (cur)
165  {
166  if ((sym==cur->function()) &&
167  reinterpret_cast<_aterm_appl<Term>*>(cur)->arg[0] == arg0)
168  {
169  return cur;
170  }
171  cur = cur->next();
172  }
173 
174  cur = detail::allocate_term(TERM_SIZE_APPL(1));
175  /* Delay masking until after allocate_term */

177

179  new (&(reinterpret_cast<detail::_aterm_appl<Term>*>(const_cast<detail::_aterm*>(cur))->arg[0])) Term(arg0);
180  insert_in_hashtable(cur,hnr);

182  call_creation_hook(cur);
