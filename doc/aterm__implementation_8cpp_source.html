161 /* Remove terms, but do not use the stack, because
162  * the stack is not always sufficiently large, esp. if limit stacksize
163  * is not set. On OSX the stack can only be 65Mbyte big, which is not enough

179 void resize_aterm_hashtable()
180 {
181  static bool resizing_aterm_hashtable_has_failed=false;
182  if (resizing_aterm_hashtable_has_failed)
183  {
184  // Not increasing the hashtable has only a slight performance penalty,
185  // as the hashtables get fuller. But it saves memory, and does not lead
186  // to incorrect behaviour.
187  return;
188  }
189  const std::size_t old_size=aterm_table_size;
190  aterm_table_size <<=1; // Double the size.
191  // Intentionally do not throw the old hashtable away before allocating the new one.
192  // It is better when the extra memory is used for blocks of aterms, than for increasing the
193  // hashtable.
194  _aterm* * new_hashtable=reinterpret_cast<_aterm**>(calloc(aterm_table_size,sizeof(_aterm*)));
195 
196  if (new_hashtable==nullptr)
197  {
198  resizing_aterm_hashtable_has_failed=true;
199  mCRL2log(mcrl2::log::warning) << "could not resize hashtable to size " << aterm_table_size << ". ";
200  aterm_table_size = old_size;
201  return;
202  }
203  aterm_table_mask = aterm_table_size-1;
204 
205  /* Rehash all old elements */
206  for (std::size_t p=0; p<old_size; ++p)
207  {
208  _aterm* aterm_walker=aterm_hashtable[p];
209 
210  while (aterm_walker)
211  {
212  assert(!aterm_walker->reference_count_indicates_is_in_freelist());
213  _aterm* next = aterm_walker->next();
214  const std::size_t hnr = hash_number(aterm_walker) & aterm_table_mask;
215  aterm_walker->set_next(new_hashtable[hnr]);
216  new_hashtable[hnr] = aterm_walker;
217  assert(aterm_walker->next()!=aterm_walker);
218  aterm_walker = next;
219  }
220  }
221  free(aterm_hashtable);
222  aterm_hashtable=new_hashtable;
223 }

225 void collect_terms_with_reference_count_0()
226 {
227  // This function puts all with reference count==0 in the freelist, in the reverse order as
228  // the sequence of blocks.
229 
230 
231  // First put all terms with reference count 0 in the freelist.
232  for(std::size_t size=TERM_SIZE; size<terminfo_size; ++size)
233  {
234  TermInfo& ti=terminfo[size];
235 
236  for(Block* b=ti.at_block; b!=nullptr; b=b->next_by_size)
237  {
238  for(std::size_t *p=b->data; p<b->end; p=p+size)
239  {
240  _aterm* p1=reinterpret_cast<_aterm*>(p);
241  if (p1->reference_count()==0)
242  {
243  // Put term in freelist, freeing subterms also.
244  free_term(p1);
245  }
246  }
247  }
248  }
249 
250  // Reconstruct the freelists for all terms, freeing empty blocks.
251  std::size_t number_of_blocks=0;
252  for(std::size_t size=TERM_SIZE; size<terminfo_size; ++size)
253  {
254  TermInfo& ti=terminfo[size];
255  Block* previous_block=nullptr;
256  ti.at_freelist=nullptr;
257  for(Block* b=ti.at_block; b!=nullptr; )
258  {
259  Block* next_block=b->next_by_size;
260  bool block_is_empty_up_till_now=true;
261  _aterm* freelist_of_previous_block=ti.at_freelist;
262  for(std::size_t *p=b->data; p<b->end; p=p+size)
263  {
264  _aterm* p1=reinterpret_cast<_aterm*>(p);
265  assert(p1->reference_count()!=0);
266  if (p1->reference_count_indicates_is_in_freelist())
267  {
268  p1->set_next(ti.at_freelist);
269  ti.at_freelist=p1;
270  }
271  else
272  {
273  block_is_empty_up_till_now=false;
274  }
275  }
276 
277  if (block_is_empty_up_till_now)
278  {
279  ti.at_freelist=freelist_of_previous_block;
280  if (previous_block==nullptr)
281  {
282  ti.at_block=next_block;
283  }
284  else
285  {
286  previous_block->next_by_size=next_block;
287  }
288  free(b);
289  }
290  else
291  {
292  previous_block=b;
293  number_of_blocks++;
294  }
295  b=next_block;
296  }
297  }
298  garbage_collect_count_down=(1+number_of_blocks)*(BLOCK_SIZE/(sizeof(std::size_t)*16));
299 }

331 void initialise_aterm_administration()
332 {
333  /* Check for reasonably sized aterm (at least 32 bits, 4 bytes). This check might break on
334  * perfectly valid architectures that have char == 2 bytes, and sizeof(header_type) == 2 */
335  static_assert(sizeof(std::size_t) == sizeof(aterm*) && sizeof(std::size_t) >= 4,"pointers and std::size_t must be equal and larger than four bytes for the aterm library");
336 
337  detail::function_adm.initialise_function_symbols();
338 
339  /* Explict initialisation on first use. This first use is when a function symbol is created for
340  * the first time, which may be due to the initialisation of a global variable in a .cpp file, or
341  * due to the initialisation of a pre-main initialisation of a static variable, which some
342  * compilers do. */
343 
344  aterm_hashtable=reinterpret_cast<_aterm**>(calloc(aterm_table_size,sizeof(_aterm*)));
345  if (aterm_hashtable==nullptr)
346  {
347  throw std::runtime_error("Out of memory. Cannot create an aterm symbol hashtable.");
348  }
349 
350  terminfo=reinterpret_cast<TermInfo*>(malloc(terminfo_size*sizeof(TermInfo)));
351  if (terminfo==nullptr)
352  {
353  throw std::runtime_error("Out of memory. Failed to allocate the terminfo array.");
354  }
355 
356  for(std::size_t i=TERM_SIZE; i<terminfo_size; ++i)
357  {
358  new (&terminfo[i]) TermInfo();
359  }
360 
361  /* Check at exit that all function symbols and terms have been cleaned up properly.
362  * TODO: on windows it turns out that the reference counts do not reduce to 0. The reason for it
363  * is unclear. It could either be due to an unforeseen sequence of destroying static and
364  * global variables, in relation to the execution of the exit function defined below. Or it
365  * could be that on windows global variables are not properly cleaned up. This requires
366  * further investigation. */
367 #ifdef MCRL2_CHECK_ATERMPP_CLEANUP
368  assert(atexit(check_that_all_objects_are_free) == 0);
369 #endif
370 
371  detail::initialise_function_map_administration();
372 
373  aterm::static_undefined_aterm=detail::term_appl0(detail::function_adm.AS_DEFAULT);
374  aterm::static_undefined_aterm->increase_reference_count();
375  aterm::static_empty_aterm_list=detail::term_appl0(detail::function_adm.AS_EMPTY_LIST);
376  aterm::static_empty_aterm_list->increase_reference_count();
377 
378 }

380 /* allocate a block of memory to contain terms consisting of `size' objects
381  * of type std::size_t or pointer */
382 void allocate_block(const std::size_t size)
383 {
384  const std::size_t block_header_size=sizeof(struct Block*)+sizeof(std::size_t*);
385  std::size_t number_of_terms_in_data_block=(BLOCK_SIZE-block_header_size) / (size*sizeof(std::size_t));
386  if (number_of_terms_in_data_block==0) number_of_terms_in_data_block=1; // Take care that there is room for at least one term.
387 
388  Block* newblock = (Block*)malloc(block_header_size+number_of_terms_in_data_block*size*sizeof(std::size_t));
389  if (newblock == nullptr)
390  {
391  throw std::runtime_error("Out of memory. Could not allocate a block of memory to store terms.");
392  }
393 
394  assert(size>=TERM_SIZE);
395  assert(size < terminfo_size);
396  TermInfo& ti = terminfo[size];
397 
398  newblock->end = newblock->data + number_of_terms_in_data_block*size;
399 
400  // Put new terms in the block in the freelist.
401 
402  for(std::size_t *p=newblock->data; p<newblock->end; p=p+size)
403  {
404  _aterm* p1=reinterpret_cast<_aterm*>(p);
405  p1->set_next(ti.at_freelist);
406  ti.at_freelist = p1;
407  p1->set_reference_count_indicates_in_freelist(false);
408  }
409 
410  newblock->next_by_size = ti.at_block;
411  ti.at_block = newblock;
412  assert(ti.at_block != nullptr);
413  assert(ti.at_freelist != nullptr);
414 }
