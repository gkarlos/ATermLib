81 {
82  int buf[8];
83 
84  // Try to read 1st character
85  if ((buf[0] = is.get()) == EOF)
86  {
87  throw std::runtime_error("Fail to read an int from the input");
88  }
89 
90  // Check if 1st character is enough
91  if ((buf[0] & 0x80) == 0)
92  {
93  return buf[0];
94  }
95 
96  // Try to read 2nd character
97  if ((buf[1] = is.get()) == EOF)
98  {
99  throw std::runtime_error("Fail to read an int from the input");
100  }
101 
102  // Check if 2nd character is enough
103  if ((buf[0] & 0x40) == 0)
104  {
105  return buf[1] + ((buf[0] & ~0xc0) << 8);
106  }
107 
108  // Try to read 3rd character
109  if ((buf[2] = is.get()) == EOF)
110  {
111  throw std::runtime_error("Fail to read an int from the input");
112  }
113 
114  // Check if 3rd character is enough
115  if ((buf[0] & 0x20) == 0)
116  {
117  return buf[2] + (buf[1] << 8) + ((buf[0] & ~0xe0) << 16);
118  }
119 
120  // Try to read 4th character
121  if ((buf[3] = is.get()) == EOF)
122  {
123  throw std::runtime_error("Fail to read an int from the input");
124  }
125 
126  // Check if 4th character is enough
127  if ((buf[0] & 0x10) == 0)
128  {
129  return buf[3] + (buf[2] << 8) + (buf[1] << 16) + ((buf[0] & ~0xf0) << 24);
