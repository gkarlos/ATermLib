33 inline std::size_t approximatepowerof2(std::size_t n)
34 {
35  std::size_t mask = n;
36 
37  while (mask >>= 1)
38  {
39  n |= mask;
40  }
41 
42  if (n<127)
43  {
44  n=127;
45  }
46  return n;
47 }

62 template <class ELEMENT>
63 inline
64 std::size_t indexed_set<ELEMENT>::put_in_hashtable(const ELEMENT& key, std::size_t n)
65 {
66  /* Find a place to insert key,
67  and find whether key already exists */
68 
69  std::size_t deleted_position=detail::EMPTY; // This variable recalls a proper deleted position to insert n. EMPTY means not yet found.
70  std::size_t start = std::hash<aterm>()(key) & sizeMinus1;
71  std::size_t c = start;
72 
73  while (true)
74  {
75  assert(c!=detail::EMPTY);
76  std::size_t v = hashtable[c];
77  assert(v==detail::EMPTY || v == detail::DELETED || v<m_keys.size());
78  if (v == detail::EMPTY)
79  {
80  /* Found an empty spot, insert a new index belonging to key,
81  preferably at a deleted position, if that has been encountered. */
82  if (deleted_position==detail::EMPTY)
83  {
84  --nr_of_insertions_until_next_rehash;
85  assert(nr_of_insertions_until_next_rehash!=npos);
86  hashtable[c] = n;
87  }
88  else
89  {
90  hashtable[deleted_position] = n;
91  }
92  return n;
93  }
94 
95  if (v == detail::DELETED)
96  {
97  /* Recall this position to be used, in case the element is not found. */
98  if (deleted_position==detail::EMPTY)
99  {
100  deleted_position=c;
101  }
102  }
103  else if (m_keys[v]==key)
104  {
105  /* key is already in the set, return position of key */
106  return v;
107  }
108  c = (c + detail::STEP) & sizeMinus1;
109  assert(c!=start); // In this case the hashtable is full, which should never happen.
110  }
111  return c;
112 }

115 template <class ELEMENT>
116 inline void indexed_set<ELEMENT>::resize_hashtable()
117 {
118  /* First determine the largest index in use */
119  std::size_t largest_used_index=0;
120  for (std::size_t i=0; i<m_keys.size(); i++)
121  {
122  ELEMENT t = m_keys[i];
123  if (t.defined() && i>largest_used_index)
124  {
125  largest_used_index=i;
126  }
127  }
128  m_keys.resize(largest_used_index+1);
129 
130  std::size_t newsizeMinus1 = detail::calculateNewSize(sizeMinus1,largest_used_index, max_load);
131 
132  hashtable.clear();
133  hashtable.resize(newsizeMinus1+1,detail::EMPTY);
134 
135  sizeMinus1=newsizeMinus1;
136  nr_of_insertions_until_next_rehash = ((sizeMinus1/100)*max_load);
137 
138 
139 
140 
141  free_positions=std::stack < std::size_t >();
142  /* rebuild the hashtable again, and put free indices in the free_position stack.
143  Count down, such that the lowest indices are highest in the stack, to be
144  re-used first. */
145  for (std::size_t i=m_keys.size(); i>0 ; )
146  {
147  --i;
148  ELEMENT t = m_keys[i];
149  if (t.defined())
150  {
151  put_in_hashtable(t, i);
152  }
153  else
154  {
155  free_positions.push(i);
156  }
157  }
158 }

170 template <class ELEMENT>
171 inline ssize_t indexed_set<ELEMENT>::index(const ELEMENT& elem) const
172 {
173  std::size_t start = std::hash<aterm>()(elem) & sizeMinus1;
174  std::size_t c = start;
175  do
176  {
177  std::size_t v=hashtable[c];
178  if (v == detail::EMPTY)
179  {
180  return npos; /* Not found. */
181  }
182  assert(v == detail::DELETED || v<m_keys.size());
183  if (v != detail::DELETED && elem==m_keys[v])
184  {
185  return v;
186  }
187 
188  c = (c+detail::STEP) & sizeMinus1;
189  assert(c!=start); // The hashtable is full. This should never happen.
190  }
191  while (true);
192 
193  return npos; /* Not found. */

196 template <class ELEMENT>
197 bool indexed_set<ELEMENT>::erase(const ELEMENT& key)
198 {
199  std::size_t start = std::hash<aterm>()(key) & sizeMinus1;
200  std::size_t c = start;
201  std::size_t v;
202  while (true)
203  {
204  v = hashtable[c];
205  if (v == detail::EMPTY)
206  {
207  return false;
208  }
209  assert(v == detail::DELETED || v<m_keys.size());
210  if (v != detail::DELETED && key==m_keys[v])
211  {
212  break;
213  }
214 
215  c = (c + detail::STEP) & sizeMinus1;
216  if (c == start)
217  {
218  assert(0); // The hashtable is full. This should never happen.
219  return false;
220  }
221  }
222 
223  hashtable[c] = detail::DELETED;
224 
225  assert(m_keys.size()>v);
226  assert(!ELEMENT().defined());
227  assert(v<m_keys.size());
228  m_keys[v]=ELEMENT();
229  free_positions.push(v);
230 
231  return true;
232 }

258 inline std::pair<std::size_t, bool> indexed_set<ELEMENT>::put(const ELEMENT& key)
259 {
260  const std::size_t m=(free_positions.empty()? m_keys.size() : free_positions.top());
261  const std::size_t n = put_in_hashtable(key,m);
262  if (n != m) // Key already exists.
263  {
264  return std::make_pair(n,false);
265  }
266 
267  if (!free_positions.empty())
268  {
269  free_positions.pop();
270  }
271  else if (n>=m_keys.size())
272  {
273  m_keys.resize(n+1);
274  }
275  assert(m_keys.size()>n);
276  m_keys[n]=key;

279  resize_hashtable();

281 
282  return std::make_pair(n, true);
