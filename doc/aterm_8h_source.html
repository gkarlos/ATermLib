91  void copy_term(const aterm& t) noexcept
92  {
93  /* It is important that the reference count of m_term is decreased after
94  the reference count of t.m_term is increased, as otherwise if the terms are exactly
95  the same, the reference count can temporarily become 0. */
96  const_cast<aterm&>(t).increase_reference_count<true>();
97  m_term->decrease_reference_count();
98  m_term=t.m_term;
