134 inline _aterm* term_appl0(const function_symbol& sym)
135 {
136  assert(sym.arity()==0);
137 
138  const std::hash<function_symbol> function_symbol_hasher;
139  std::size_t hnr = function_symbol_hasher(sym);
140 
141  _aterm *cur = detail::aterm_hashtable[hnr & detail::aterm_table_mask];
142 
143  while (cur)
